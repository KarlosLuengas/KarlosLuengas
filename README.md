      let pos = 0;
      let direccion = 1;

      function animar() {
        if (pos >= 300) direccion = -1;
        if (pos <= 0) direccion = 1;
        pos += direccion * 2;
        animarDivs.forEach(div => {
          div.style.left = pos + "px";
        });
        requestAnimationFrame(animar);
      }

      animar();
